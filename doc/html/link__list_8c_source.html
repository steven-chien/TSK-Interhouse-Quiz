91 void listRemove(node *q, list *theList) {
92 
93  node *p;
94  int i;
95  if(q==theList->start) {
96  theList->start = theList->start->next;
97  wprintw(msg_content, "remove from start\n");
98  wrefresh(msg_content);
99  }
100  else {
101  for(i=0, p=theList->start; i<theList->size; i++,p=p->next) {
102  if(p->next==q) {
103  if(theList->end==q) {
104  theList->end = p;
105  }
106  else {
107  p->next = q->next;
108  }
109  break;
110  }
111  }
112  wprintw(msg_content, "remove from tail\n");
113  wrefresh(msg_content);
114  }
115  theList->size--;
116  if(theList->size==0) {
117  theList->start = theList->end = NULL;
118  }
119  free(q);
120 }
